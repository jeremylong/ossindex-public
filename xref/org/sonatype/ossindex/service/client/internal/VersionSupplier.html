13  package org.sonatype.ossindex.service.client.internal;
14  
15  import com.google.common.base.Supplier;
16  
17  /**
18   * {@link Version} supplier.
19   *
20   * @since ???
21   */
22  public class VersionSupplier
23      implements Supplier<Version>
24  {
25    private static volatile Version instance;
26  
27    @Override
28    public Version get() {
29      if (instance == null) {
30        instance = new Version(VersionSupplier.class);
31      }
32      return instance;
33    }
34  }
