13  package org.sonatype.ossindex.service.client;
14  
15  import java.net.URI;
16  
17  import javax.annotation.Nullable;
18  
19  import com.fasterxml.jackson.annotation.JsonProperty;
20  import com.google.common.cache.CacheBuilderSpec;
21  
22  /**
23   * {@link OssindexClient} configuration.
24   *
25   * @since ???
26   */
27  public class OssindexClientConfiguration
28  {
29    public static final URI DEFAULT_BASE_URL = URI.create("https://ossindex.sonatype.org/");
30  
31    public static final int DEFAULT_BATCH_SIZE = 64;
32  
33    public static final CacheBuilderSpec DEFAULT_CACHE = CacheBuilderSpec.parse(
34        "maximumSize=256,expireAfterAccess=2m,softValues"
35    );
36  
37    @JsonProperty
38    private URI baseUrl = DEFAULT_BASE_URL;
39  
40    @JsonProperty
41    private int batchSize = DEFAULT_BATCH_SIZE;
42  
43    @JsonProperty
44    private CacheBuilderSpec reportCache = DEFAULT_CACHE;
45  
46    @Nullable
47    @JsonProperty("auth")
48    private AuthConfiguration authConfiguration;
49  
50    @Nullable
51    @JsonProperty("proxy")
52    private ProxyConfiguration proxyConfiguration;
53  
54    public URI getBaseUrl() {
55      return baseUrl;
56    }
57  
58    public void setBaseUrl(URI baseUrl) {
59      this.baseUrl = baseUrl;
60    }
61  
62    public void setBaseUrl(final String baseUrl) {
63      setBaseUrl(URI.create(baseUrl));
64    }
65  
66    public int getBatchSize() {
67      return batchSize;
68    }
69  
70    public void setBatchSize(int batchSize) {
71      this.batchSize = batchSize;
72    }
73  
74    public CacheBuilderSpec getReportCache() {
75      return reportCache;
76    }
77  
78    public void setReportCache(CacheBuilderSpec reportCache) {
79      this.reportCache = reportCache;
80    }
81  
82    @Nullable
83    public AuthConfiguration getAuthConfiguration() {
84      return authConfiguration;
85    }
86  
87    public void setAuthConfiguration(@Nullable final AuthConfiguration authConfiguration) {
88      this.authConfiguration = authConfiguration;
89    }
90  
91    @Nullable
92    public ProxyConfiguration getProxyConfiguration() {
93      return proxyConfiguration;
94    }
95  
96    public void setProxyConfiguration(@Nullable final ProxyConfiguration proxyConfiguration) {
97      this.proxyConfiguration = proxyConfiguration;
98    }
99  }
