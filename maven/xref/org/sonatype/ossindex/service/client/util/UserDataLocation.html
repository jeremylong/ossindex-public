13  package org.sonatype.ossindex.service.client.util;
14  
15  import java.nio.file.Files;
16  import java.nio.file.Path;
17  import java.nio.file.Paths;
18  import java.util.Locale;
19  
20  import javax.annotation.Nullable;
21  
22  import com.google.common.annotations.VisibleForTesting;
23  import com.google.common.base.Supplier;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import static com.google.common.base.Preconditions.checkNotNull;
28  
29  /**
30   * Helper to return user data base directory location specific to {@link OsType}.
31   *
32   * @since ???
33   */
34  public class UserDataLocation
35      implements Supplier<Path>
36  {
37    private static final Logger log = LoggerFactory.getLogger(UserDataLocation.class);
38  
39    private final String vendor;
40  
41    private final String product;
42  
43    public UserDataLocation(final String vendor, final String product) {
44      this.vendor = checkNotNull(vendor);
45      this.product = checkNotNull(product);
46    }
47  
48    @Override
49    public Path get() {
50      OsType osType = osType();
51      log.debug("OS type: {}", osType);
52  
53      switch (osType) {
54        case MAC:
55          return mac();
56  
57        case WINDOWS:
58          return windows();
59  
60        case LINUX:
61          return linux();
62  
63        case OTHER:
64        default:
65          return other();
66      }
67    }
68  
69    @VisibleForTesting
70    protected OsType osType() {
71      return OsType.detect();
72    }
73  
74    @VisibleForTesting
75    @Nullable
76    protected String getenv(final String name) {
77      return System.getenv(name);
78    }
79  
80    @VisibleForTesting
81    protected Path userHome() {
82      return Paths.get(System.getProperty("user.home"));
83    }
84  
85    /**
86     * Return user-data location for Mac OS.
87     */
88    private Path mac() {
89      return userHome().resolve(Paths.get("Library", "Application Support", vendor, product));
90    }
91  
92    /**
93     * Return user-data location for Linux.
94     */
95    private Path linux() {
96      String dirName = String.format("%s-%s", vendor, product).toLowerCase(Locale.ENGLISH);
97      return userHome().resolve(Paths.get(".config", dirName));
98    }
99  
100   /**
101    * Return fallback user-data location for others.
102    */
103   private Path other() {
104     String dirName = String.format(".%s-%s", vendor, product).toLowerCase(Locale.ENGLISH);
105     return userHome().resolve(dirName);
106   }
107 
108   //
109   // Windows
110   //
111 
112   /**
113    * Environment variable for modern (vista+) Windows user-data location base-directory.
114    */
115   @VisibleForTesting
116   static final String LOCALAPPDATA = "LOCALAPPDATA";
117 
118   /**
119    * Environment variable for legacy (<=XP) Windows user-data location base-directory.
120    */
121   @VisibleForTesting
122   static final String APPDATA = "APPDATA";
123 
124   @VisibleForTesting
125   static final String USERPROFILE = "USERPROFILE";
126 
127   @Nullable
128   private Path windowsBaseDir() {
129     // modern windows; vista,7+
130     String dirName = getenv(LOCALAPPDATA);
131     if (dirName != null) {
132       return Paths.get(dirName);
133     }
134 
135     // windows xp and older?
136     dirName = getenv(APPDATA);
137     if (dirName != null) {
138       return Paths.get(dirName);
139     }
140 
141     // before giving up; try what APPDATA should be; and if it exists use it
142     dirName = getenv(USERPROFILE);
143     if (dirName != null) {
144       Path path = Paths.get(dirName, "Local Settings", "Application Data");
145       if (Files.exists(path)) {
146         return path;
147       }
148     }
149 
150     return null;
151   }
152 
153   /**
154    * Return user-data location for Windows.
155    */
156   private Path windows() {
157     Path basedir = windowsBaseDir();
158     if (basedir != null) {
159       return basedir.resolve(Paths.get(vendor, product));
160     }
161 
162     // fallback
163     log.warn("Unable to determine user-data location for Windows");
164     return other();
165   }
166 }
