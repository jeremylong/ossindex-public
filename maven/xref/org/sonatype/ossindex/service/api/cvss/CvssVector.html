15  import java.util.Collections;
16  import java.util.HashMap;
17  import java.util.LinkedHashMap;
18  import java.util.LinkedList;
19  import java.util.List;
20  import java.util.Map;
21  
22  import javax.annotation.Nullable;
23  
24  import com.google.common.base.MoreObjects;
25  import com.google.common.base.Splitter;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import static com.google.common.base.Preconditions.checkNotNull;
30  import static org.sonatype.ossindex.service.api.cvss.CvssMetric.UNKNOWN_GROUP;
31  
32  /**
33   * CVSS vector.
34   *
35   * @since 1.2.0
36   */
37  public abstract class CvssVector
38  {
39    private static final Logger log = LoggerFactory.getLogger(CvssVector.class);
40  
41    private final String version;
42  
43    private final String value;
44  
45    private final Map<String, String> metrics;
46  
47    public CvssVector(final String version, final String value) {
48      this.version = checkNotNull(version);
49      this.value = checkNotNull(value);
50      this.metrics = Collections.unmodifiableMap(parse(value));
51    }
52  
53    public String getVersion() {
54      return version;
55    }
56  
57    public String getValue() {
58      return value;
59    }
60  
61    public Map<String, String> getMetrics() {
62      return metrics;
63    }
64  
65    private static final Splitter SLASH_SPLITTER = Splitter.on('/').trimResults();
66  
67    private static final Splitter COLON_SPLITTER = Splitter.on(':').trimResults();
68  
69    private Map<String, String> parse(final String value) {
70      Map<String, String> result = new LinkedHashMap<>();
71  
72      for (String metric : SLASH_SPLITTER.split(value)) {
73        List<String> parts = COLON_SPLITTER.splitToList(metric);
74        result.put(parts.get(0), parts.get(1));
75      }
76      return result;
77    }
78  
79    public List<MetricDescription> getDescriptions() {
80      List<MetricDescription> result = new LinkedList<>();
81      for (Map.Entry<String, String> entry : metrics.entrySet()) {
82        String metric = entry.getKey();
83        result.add(describe(metric, entry.getValue()));
84      }
85      return result;
86    }
87  
88    @Nullable
89    protected abstract CvssMetric findMetric(final String code);
90  
91    protected MetricDescription describe(final String metricCode, final String metricValue) {
92      String group = UNKNOWN_GROUP;
93      String title = metricCode;
94      String meaning = metricValue;
95  
96      CvssMetric metric = findMetric(metricCode);
97      if (metric != null) {
98        group = metric.getGroup();
99        title = metric.getTitle();
100       meaning = metric.getMeaning(metricValue);
101 
102       // cope with undefined meanings
103       if (meaning == null) {
104         meaning = metricValue;
105       }
106     }
107 
108     if (metricCode.equals(title) || metricValue.equals(meaning)) {
109       log.warn("Did not translate; metric: {} -> {}; meaning: {} -> {}", metricCode, title, metricValue, meaning);
110     }
111 
112     return new MetricDescription(metricCode, group, title, metricValue, meaning);
113   }
114 
115   @Override
116   public String toString() {
117     return MoreObjects.toStringHelper(this)
118         .add("version", version)
119         .add("value", value)
120         .add("metrics", metrics)
121         .toString();
122   }
123 
124   protected static Map<String, CvssMetric> createLookup(final CvssMetric[] metrics) {
125     Map<String, CvssMetric> lookup = new HashMap<>(metrics.length);
126     for (CvssMetric metric : metrics) {
127       CvssMetric prev = lookup.put(metric.getCode(), metric);
128       if (prev != null) {
129         log.warn("Duplicate metric code: {}", metric.getCode());
130       }
131     }
132     return Collections.unmodifiableMap(lookup);
133   }

136   // MetricDescription

138 
139   public static class MetricDescription
140   {
141     private final String code;
142 
143     private final String group;
144 
145     private final String title;
146 
147     private final String valueCode;
148 
149     private final String value;
150 
151     public MetricDescription(final String code,
152                              final String group,
153                              final String title,
154                              final String valueCode,
155                              final String value)
156     {
157       this.code = code;
158       this.group = group;
159       this.title = title;
160       this.valueCode = valueCode;
161       this.value = value;
162     }
163 
164     public String getCode() {
165       return code;
166     }
167 
168     public String getGroup() {
169       return group;
170     }
171 
172     public String getTitle() {
173       return title;
174     }
175 
176     public String getDisplayTitle() {
177       if (title.equals(code)) {
178         return title;
179       }
180       return String.format("%s (%s)", title, code);
181     }
182 
183     public String getValueCode() {
184       return valueCode;
185     }
186 
187     public String getValue() {
188       return value;
189     }
190 
191     public String getDisplayValue() {
192       if (value.equals(valueCode)) {
193         return value;
194       }
195       return String.format("%s (%s)", value, valueCode);
196     }
197 
198     @Override
199     public String toString() {
200       return MoreObjects.toStringHelper(this)
201           .add("code", code)
202           .add("group", group)
203           .add("title", title)
204           .add("valueCode", valueCode)
205           .add("value", value)
206           .toString();
207     }
208   }
