13  package org.sonatype.ossindex.service.api.componentreport;
14  
15  import java.util.List;
16  
17  import javax.ws.rs.Consumes;
18  import javax.ws.rs.GET;
19  import javax.ws.rs.POST;
20  import javax.ws.rs.Path;
21  import javax.ws.rs.PathParam;
22  import javax.ws.rs.Produces;
23  
24  import org.sonatype.goodies.packageurl.PackageUrl;
25  
26  import io.swagger.annotations.Api;
27  import io.swagger.annotations.ApiOperation;
28  import io.swagger.annotations.ApiParam;
29  import io.swagger.annotations.ApiResponse;
30  import io.swagger.annotations.ApiResponses;
31  import io.swagger.annotations.Authorization;
32  
33  import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
34  import static org.sonatype.ossindex.service.api.componentreport.ComponentReportMediaTypes.REPORT_V1_JSON;
35  import static org.sonatype.ossindex.service.api.componentreport.ComponentReportMediaTypes.REQUEST_V1_JSON;
36  
37  /**
38   * {@link ComponentReport} endpoint.
39   *
40   * @since 1.0.0
41   */
42  @Path("/api/v3/component-report")
43  @Api(
44      value = "Component vulnerability reports",
45      authorizations = {
46          @Authorization("basicAuth"),
47          @Authorization("apiToken")
48      }
49  )
50  public interface ComponentReportEndpoint
51  {
52    @GET
53    @Path("{coordinates:.*}")
54    @Produces({REPORT_V1_JSON, APPLICATION_JSON})
55    @ApiOperation(value = "Request vulnerability report for component")
56    @ApiResponses({
57        @ApiResponse(code = 200, message = "Vulnerability report for component"),
58        @ApiResponse(code = 404, message = "Component not found"),
59        @ApiResponse(code = 400, message = "Missing coordinates version"),
60        @ApiResponse(code = 429, message = "Too many requests")
61    })
62    @SuppressWarnings("RestParamTypeInspection")
63    ComponentReport get(@PathParam("coordinates") @ApiParam("Coordinates as package-url") PackageUrl coordinates);
64  
65    @POST
66    @Consumes({REQUEST_V1_JSON, APPLICATION_JSON})
67    @Produces({REPORT_V1_JSON, APPLICATION_JSON})
68    @ApiOperation(value = "Request vulnerability reports for components")
69    @ApiResponses({
70        @ApiResponse(code = 200, message = "Vulnerability report for components"),
71        @ApiResponse(code = 400, message = "Request for too many components"),
72        @ApiResponse(code = 400, message = "One or more coordinates required"),
73        @ApiResponse(code = 400, message = "Missing coordinates version"),
74        @ApiResponse(code = 429, message = "Too many requests")
75    })
76    List<ComponentReport> post(ComponentReportRequest request);
77  }
